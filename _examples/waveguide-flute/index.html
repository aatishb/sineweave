--- layout: livecode title: Flute (WaveGuide) category: instrument blurb: | A simplified physical waveguide model of a flute, based on the model by Perry Cook described [here](https://ccrma.stanford.edu/software/clm/compmus/clm-tutorials/pm.html#s-f) and [here](http://quod.lib.umich.edu/cgi/p/pod/dod-idx?c=icmc;idno=bbp2372.1992.072). Here the frequency is adjustable. --- let output, flute, emb, noise, env; let freq = 700; let r1 = 0.5, r2 = 0.66, cutoff = 0.25, breath = 0.5; let lpfilter; let dcfilter; function setup() { lpfilter = lowPass(cutoff); dcfilter = dcBlocker(0.005); slider('cutoff', 0, 1, 0.01); slider('breath', 0, 1, 0.01); slider('r1', 0, 1, 0.01); slider('r2', 0, 1, 0.01); let waveGuideLength = Math.round(sampleRate / freq - 0.5); flute = new WaveGuide(waveGuideLength); emb = new WaveGuide(waveGuideLength); output = new Wave(); } function loop() { noise = noiseWave().mult(breath); emb.set(flute.get().mult(r1).add(noise)); emb.delay(numSamples); emb.set(emb.get() .map(e => Math.tanh(e)) .map(dcfilter.apply) ); flute.set(flute.get().mult(r2).add(emb.get())); lpfilter.set(cutoff); output = flute.get().map(lpfilter.apply); flute.set(output); flute.delay(numSamples); return output.mult(0.1); }
