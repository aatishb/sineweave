--- layout: livecode title: Flute category: instrument blurb: | A simplified physical waveguide model of a flute, based on the model by Perry Cook described [here](https://ccrma.stanford.edu/software/clm/compmus/clm-tutorials/pm.html#s-f) and [here](http://quod.lib.umich.edu/cgi/p/pod/dod-idx?c=icmc;idno=bbp2372.1992.072). --- let output, flute, emb, noise, env; let freq = 700; let r1 = 0.66, r2 = 0.66, cutoff = 0.5, breath = 0.5; let lpfilter; let dcfilter; function setup() { lpfilter = lowPass(cutoff); dcfilter = dcBlocker(0.005); slider('breath', 0, 1, 0.01); slider('r1', 0, 1, 0.01); slider('r2', 0, 1, 0.01); slider('cutoff', 0, 1, 0.01); let waveTableLength = Math.round(sampleRate / freq - 0.5); flute = new WaveTable(waveTableLength); emb = new WaveTable(waveTableLength); output = new Wave(); } function loop() { noise = time.map(whiteNoise).mult(breath); // env = time.map(adsr(0, 0.5, 0.5, 0.5, 0.5)); // noise = noise.mult(env); emb.update(flute.get().mult(r1).add(noise)); emb.delay(128); emb.update(emb.get() .map(e => Math.tanh(e)) .map(dcfilter.apply) ); flute.update(flute.get().mult(r2).add(emb.get())); lpfilter.set(cutoff); output = flute.get().map(lpfilter.apply); flute.update(output); flute.delay(128); return output.mult(0.1); }
