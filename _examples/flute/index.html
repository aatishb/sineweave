--- layout: livecode title: Flute category: waveguide blurb: | A simplified physical waveguide model of a flute, based on the model by Perry Cook. --- let output, flute, emb, noise, env; let freq = 700; let r1 = 0.66, r2 = 0.66, cutoff = 0.5, breath = 0.5; let filter = new lowPass(cutoff); let dcfilter = new dcBlocker(0.005); function setup() { slider('breath', 0, 1, 0.01); slider('r1', 0, 1, 0.01); slider('r2', 0, 1, 0.01); slider('cutoff', 0, 1, 0.01); let waveTableLength = Math.round(sampleRate / freq - 0.5); flute = new WaveTable(waveTableLength); emb = new WaveTable(waveTableLength); output = new Float32Array(128); } function loop() { noise = time.map(whiteNoise).mult(breath); // env = time.map(adsr(0, 0.5, 0.5, 0.5, 0.5)); // noise = noise.convolve(env); emb.update(flute.get().mult(r1).add(noise)); emb.delay(128); emb.update(emb.get() .map(e => Math.tanh(e)) .applyFilter(dcfilter) ); flute.update(flute.get().mult(r2).add(emb.get())); filter.set(cutoff); output = flute.get().applyFilter(filter); flute.update(output); flute.delay(128); return output.mult(0.1); }
