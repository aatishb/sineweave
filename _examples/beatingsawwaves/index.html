--- layout: livecode title: Space Drone (Beating Saw Waves) category: synthesis --- // based on example by Jean Claude Risset // "Computer Music: Why?" WERGO 2033-2, 1985 // https://www.kadenze.com/courses/physics-based-sound-synthesis-for-games-and-interactive-systems-iv/sessions/physics-oscillators-sines-spectra-spectral-additive-synthesis let maxIndex = 10; let mySum; let zeroWave, myWave; let freq; function setup() { zeroWave = new Wave(); freq = new Wave(maxIndex); let baseFreq = 150; // play with changing this freq = freq.map((e,i) => baseFreq + 0.1 * i); // change a few of the frequencies in the array to get interesting beats freq[1] = 105.2; freq[2] = 105; freq[5] = 205; freq[6] = 207; } function loop() { myWave = zeroWave; freq.forEach(f => { myWave = myWave.add(time.map(t => sawB(f, t))); }); return myWave.mult(0.3).div(maxIndex); } function sawB(f, t) { mySum = 0; for (let k = 1; k < maxIndex; k++){ mySum += Math.sin(2 * Math.PI * k * f * t) / k; } mySum *= 1 / Math.PI; return (0.5 - mySum) * 2 - 1; }
