--- layout: livecode title: Steam Whistle (WaveGuide) category: instrument blurb: --- let output, waveGuide, noise; let freq = 500; let feedback = 1.67, breath = 0.67, cutoff = 0.3; let lpfilter, dcfilter; function setup() { slider('feedback', 0, 2, 0.01); slider('breath', 0, 1, 0.01); slider('cutoff', 0, 1, 0.01); let waveGuideLength = Math.round(sampleRate / freq - 0.5); waveGuide = new WaveGuide(waveGuideLength); lpfilter = lowPass(cutoff); dcfilter = dcBlocker(0.005); } function loop() { noise = noiseWave().mult(breath); waveGuide.set(waveGuide.get().mult(feedback).add(noise)); waveGuide.delay(numSamples); lpfilter.set(cutoff); output = waveGuide.get() .map(e => Math.tanh(e)) .map(dcfilter.apply) .map(lpfilter.apply); waveGuide.set(output); return output.mult(0.1); }
