--- layout: livecode title: Sine Wave blurb: | A sine wave whose amplitude and loudness are controlled by a slider. --- let freq = 600; let amp; let dbfs = -20; function setup() { slider('freq', 20, 1000, 1); slider('dbfs', -27, -7, 1); } function loop() { amp = Math.pow(10, dbfs/20); return time .map(sin(freq)) .mult(amp); }
