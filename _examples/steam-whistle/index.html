--- layout: livecode title: Steam Whistle category: instrument blurb: --- let output, wave, noise; let feedback = 1.67, breath = 0.67, cutoff = 0.3; let lpfilter, dcfilter; function setup() { slider('feedback', 0, 2, 0.01); slider('breath', 0, 1, 0.01); slider('cutoff', 0, 1, 0.01); wave = new Wave(); lpfilter = lowPass(cutoff); dcfilter = dcBlocker(0.005); } function loop() { lpfilter.set(cutoff); noise = noiseWave().mult(breath); wave = wave .mult(feedback) .add(noise) .map(e => Math.tanh(e)) .apply(lpfilter) .apply(dcfilter); return wave.mult(0.1); }
